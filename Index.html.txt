<script>
  // ── State ──────────────────────────────────────────────────────
  let imgB64 = null, imgMime = null;

  // ── API Key ────────────────────────────────────────────────────
  const storedKey = localStorage.getItem("mv_apikey");
  if (storedKey) {
    document.getElementById("api-key").value = storedKey;
    showSavedKey();
  }

  function onKeyInput() {
    const k = document.getElementById("api-key").value.trim();
    if (k) localStorage.setItem("mv_apikey", k);
    updateBtn();
  }

  function showSavedKey() {
    document.getElementById("key-saved-view").style.display = "block";
    document.getElementById("key-input-view").style.display = "none";
    updateBtn();
  }

  function clearKey() {
    localStorage.removeItem("mv_apikey");
    document.getElementById("api-key").value = "";
    document.getElementById("key-saved-view").style.display = "none";
    document.getElementById("key-input-view").style.display = "block";
    updateBtn();
  }

  function toggleKeyVis() {
    const inp = document.getElementById("api-key");
    const btn = document.getElementById("eye-btn");
    if (inp.type === "password") { inp.type = "text"; btn.textContent = "🙈"; }
    else { inp.type = "password"; btn.textContent = "👁"; }
  }

  function getKey() { return (localStorage.getItem("mv_apikey") || document.getElementById("api-key").value || "").trim(); }

  // ── Image ──────────────────────────────────────────────────────
  function onFileChange(e) { loadFile(e.target.files[0]); }

  function loadFile(file) {
    if (!file || !file.type.startsWith("image/")) return;
    imgMime = file.type;
    const reader = new FileReader();
    reader.onload = e => {
      imgB64 = e.target.result.split(",")[1];
      document.getElementById("preview-img").src = e.target.result;
      document.getElementById("preview-wrap").style.display = "block";
      document.getElementById("drop-zone").style.display = "none";
      updateBtn();
    };
    reader.readAsDataURL(file);
  }

  function removeImage() {
    imgB64 = null; imgMime = null;
    document.getElementById("preview-wrap").style.display = "none";
    document.getElementById("drop-zone").style.display = "block";
    document.getElementById("file-input").value = "";
    updateBtn();
  }

  // ── Sources ────────────────────────────────────────────────────
  function toggleSrc(el) {
    el.classList.toggle("on");
    el.querySelector(".src-check").textContent = el.classList.contains("on") ? "✓" : "";
  }

  function getSelectedSources() {
    return Array.from(document.querySelectorAll(".src-item.on")).map(el => ({
      name: el.querySelector(".src-name").textContent,
      url: el.dataset.url,
      paid: el.dataset.paid === "true"
    }));
  }

  // ── Button State ───────────────────────────────────────────────
  function updateBtn() {
    const btn = document.getElementById("btn");
    const ok = getKey() && imgB64;
    btn.disabled = !ok;
    btn.textContent = !getKey() ? "API-Key eingeben" : !imgB64 ? "Bild hochladen" : "Analyse starten";
  }

  // ── JSON extraction ────────────────────────────────────────────
  function extractJSON(text) {
    const fns = [
      () => JSON.parse(text.trim()),
      () => JSON.parse(text.replace(/^```json\s*/i,"").replace(/```\s*$/i,"").trim()),
      () => { const s=text.indexOf("{"),e=text.lastIndexOf("}"); if(s<0||e<=s) throw 0; return JSON.parse(text.slice(s,e+1)); }
    ];
    for (const fn of fns) { try { return fn(); } catch(_){} }
    return null;
  }

  // ── Analyze ────────────────────────────────────────────────────
  async function analyze() {
    const key = getKey();
    const sources = getSelectedSources();
    if (!key || !imgB64 || !sources.length) return;

    const btn = document.getElementById("btn");
    const output = document.getElementById("output");
    btn.disabled = true;
    output.innerHTML = '<div class="loading-box"><div class="spinner"></div>Analysiere Bild...</div>';

    const srcLines = sources.map(s => "- " + s.name + " (" + s.url + ")").join("\n");
    const paidNames = sources.filter(s=>s.paid).map(s=>s.name).join(", ");
    const specialty = document.getElementById("specialty").value;
    const ctx = document.getElementById("context").value;

    const system =
      "You are a clinical AI assistant for licensed physicians. Analyze the medical image and respond with a differential diagnosis in German.\n\n" +
      "Specialty: " + specialty + "\n\n" +
      "Sources:\n" + srcLines + "\n\n" +
      (paidNames ? "Note: " + paidNames + " need institutional login; still provide search links.\n\n" : "") +
      "CRITICAL: Respond with ONLY a raw JSON object. No markdown, no backticks, no text before or after. Start with { end with }.\n\n" +
      'Schema: {"bildqualitaet":"...","befund":"...","differentialdiagnosen":[{"rang":1,"diagnose":"...","wahrscheinlichkeit":"hoch","begruendung":"...","icd10":"..."}],"empfehlungen":"...","quellen":[{"name":"...","hinweis":"...","url":"...","kostenpflichtig":false}]}';

    try {
      const res = await fetch("https://api.anthropic.com/v1/messages", {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          "x-api-key": key,
          "anthropic-version": "2023-06-01",
          "anthropic-dangerous-direct-browser-access": "true"
        },
        body: JSON.stringify({
          model: "claude-opus-4-5",
          max_tokens: 1000,
          system,
          messages: [{
            role: "user",
            content: [
              { type: "image", source: { type: "base64", media_type: imgMime, data: imgB64 } },
              { type: "text", text: "Analyze this image. Return ONLY JSON." + (ctx ? " Context: " + ctx : "") }
            ]
          }]
        })
      });

      const data = await res.json();
      if (!res.ok) throw new Error(data.error?.message || "API Fehler " + res.status);

      const raw = (data.content||[]).map(b=>b.text||"").join("");
      const parsed = extractJSON(raw);
      if (!parsed) throw new Error("Antwort konnte nicht verarbeitet werden.");

      // Save key on success
      localStorage.setItem("mv_apikey", key);
      showSavedKey();

      renderResult(output, parsed);
    } catch(e) {
      output.innerHTML = '<div style="padding:16px"><div class="error-box"><strong>Fehler:</strong> ' + e.message + '</div></div>';
    } finally {
      btn.disabled = false;
      btn.textContent = "Erneut analysieren";
    }
  }

  // ── Render ─────────────────────────────────────────────────────
  function renderResult(container, d) {
    const rankClass = ["r1","r2","r3"];
    const probClass = {"hoch":"prob-hoch","mittel":"prob-mittel","niedrig":"prob-niedrig"};

    const diagCards = (d.differentialdiagnosen||[]).map((dx,i) => `
      <div class="diag-card ${rankClass[i]||'r3'}">
        <div class="diag-num">${dx.rang}</div>
        <div class="diag-body">
          <div class="diag-name">${dx.diagnose}${dx.icd10?` <span class="diag-icd">(${dx.icd10})</span>`:""}</div>
          <div class="diag-reason">${dx.begruendung}</div>
        </div>
        <div class="diag-prob ${probClass[dx.wahrscheinlichkeit]||'prob-niedrig'}">${dx.wahrscheinlichkeit}</div>
      </div>`).join("");

    const srcTags = (d.quellen||[]).map(q =>
      `<div class="src-tag">${q.kostenpflichtig?"🔒 ":"📄 "}<a href="${q.url}" target="_blank">${q.name}</a>${q.hinweis?" — "+q.hinweis:""}</div>`
    ).join("");

    container.innerHTML = `
      <div class="result-card">
        <div class="result-title">🖼 Bildbefund</div>
        <div class="prose">
          <p><strong style="color:#e6edf3">Bildqualität:</strong> ${d.bildqualitaet}</p>
          <p><strong style="color:#e6edf3">Befund:</strong> ${d.befund}</p>
        </div>
      </div>
      <div class="result-card">
        <div class="result-title">🎯 Differentialdiagnosen</div>
